" ==========================================================
" ~/.vimrc
" Portable Python + Git + AI terminal development environment
" ==========================================================

" ------------------ Basics ------------------
set nocompatible
filetype plugin indent on
syntax on

set number
set relativenumber
set termguicolors

set hidden
set updatetime=300
" set signcolumn=yes

" Leader
let mapleader="\\"

" ==========================================================
" Smart Autoformat + Autosave
" ==========================================================
" We format BEFORE saving to disk.

" Run Ruff fixer right before a file write
augroup ruff_format_on_write
  autocmd!
  autocmd BufWritePre *.py silent! ALEFix
augroup END

" Leaving insert mode triggers the save (which triggers formatting first)
augroup autosave_on_insertleave
  autocmd!
  autocmd InsertLeave * silent! write
augroup END

" ==========================================================
" Keymaps
" ==========================================================

" Quit / Write
nnoremap  q :q<CR>
nnoremap  w :w<CR>

" Tabs
nnoremap  t :tabnew<CR>
nnoremap  h :tabnext<CR>
nnoremap  g :tabprevious<CR>

" Window navigation
nnoremap  e <C-w>k
nnoremap  f <C-w>l
nnoremap  d <C-w>h
nnoremap  c <C-w>j

" Splits
nnoremap  s :split<CR>
nnoremap  v :vsplit<CR>

" Terminal in new tab
nnoremap  x :tabnew <Bar> terminal ++curwin<CR>

" Leave terminal mode back to Vim normal mode
tnoremap  z <C-\><C-n>

" nnn file browser in new tab
function! OpenNnnInNewTab()
  tabnew
  execute 'NnnPicker'
endfunction
nnoremap  n :call OpenNnnInNewTab()<CR>

" ==========================================================
" Plugins (vim-plug)
" ==========================================================
" Install vim-plug first if needed:
" curl -fLo ~/.vim/autoload/plug.vim --create-dirs \
" https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim

call plug#begin('~/.vim/plugged')

" Linting / formatting engine
Plug 'dense-analysis/ale'

" AI completion
Plug 'github/copilot.vim'

" File navigation
Plug 'mcchrish/nnn.vim'

" Git integration
Plug 'tpope/vim-fugitive'
Plug 'airblade/vim-gitgutter'

call plug#end()

" ==========================================================
" ALE + Ruff (uv-aware)
" ==========================================================

" Detect .venv automatically (works with uv)
let g:ale_python_auto_virtualenv = 1
let g:ale_python_ruff_executable = 'ruff'

" Use Ruff as linter
let g:ale_linters = {
\   'python': ['ruff'],
\}

" Ruff fixes: lint, organize imports, format
let g:ale_fixers = {
\   'python': ['ruff', 'ruff_format'],
\}

" We manually run fixes on write via BufWritePre
let g:ale_fix_on_save = 0

" Lint responsiveness
let g:ale_lint_on_insert_leave = 1
let g:ale_lint_on_text_changed = 'normal'

" UI behavior
let g:ale_echo_cursor = 1
let g:ale_virtualtext_cursor = 1
let g:ale_completion_enabled = 0

" ==========================================================
" Python provider
" ==========================================================
" Prevents Vim from grabbing wrong system python
let g:python3_host_prog = '/usr/bin/python3'

" ==========================================================
" Copilot
" ==========================================================
let g:copilot_enabled = 1
